# Tabs 标签页

# 基本用法

示例代码

<template>
    <t-tabs :selected="selectedTab" >
      <t-tabs-head>
        <t-tabs-item name="1" >
          标题一
        </t-tabs-item>
        <t-tabs-item name="2">
          标题二
        </t-tabs-item>
        <t-tabs-item name="3" disabled>
          标题三
        </t-tabs-item>
      </t-tabs-head>
      <t-tabs-body>
        <t-tabs-pane name="1">
          内容一
        </t-tabs-pane>
        <t-tabs-pane name="2">
          内容二
        </t-tabs-pane>
        <t-tabs-pane name="3">
          内容三
        </t-tabs-pane>
      </t-tabs-body>
    </t-tabs>
</template>

<script>
import Tabs from '../../../src/Tabs';
import TabsHead from '../../../src/Tabs-head';
import TabsBody from '../../../src/Tabs-body';
import TabsItem from '../../../src/Tabs-item';
import TabsPane from '../../../src/Tabs-pane';
import Icon from "../../../src/Icon";
export default {
  components:{
    't-tabs': Tabs,
    't-tabs-head': TabsHead,
    't-tabs-body': TabsBody,
    't-tabs-item': TabsItem,
    't-tabs-pane': TabsPane,
    't-icon':Icon,
  },
  data(){
    return {
      selectedTab:'2'
    }
  }
};
</script>
